              }
            >

              </div>
            </div>
            <Upload
              name="fdata"
              action={uploadApi}
              headers={headers}
              listType="picture-card"
              fileList={proofPics}
              onChange={(info) => changeMoreImg(info, proofPics, setProofPics)}
            >
              {proofPics.length >= 6 ? null : <div>其他（选填）</div>}
            </Upload>
          </Form.Item>
          <RenderTitle title={"保证说明"} />
          <Row>
            <Col span={7}></Col>
            <Col style={styles.info} span={11}>
              投诉人及其代理人（统称为：声明人）诚意做出如下声明：声明人知晓并确认此次投诉上传提交的所有资料及其所承载内容均真实、合法、有效，承诺对脉脉站方因此做出的所有处理措施承担一切责任。
              <div>
                <Checkbox defaultChecked={true}>

                </Checkbox>
              </div>
              <div>

                  <div style={styles.grey}>
                    以上的投诉信息将会被转发给被投诉人
                  </div>

              </div>
            </Col>
          </Row>

          <Row>
            <Col span={12}></Col>
            <Button type="primary" htmlType="submit">
              提交
            </Button>
          </Row>
        </Form>
      </div>
    </div>
  );
